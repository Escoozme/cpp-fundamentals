1.

start
	
	input balance
	input rate

	set yearly = balance * rate /100
	set semiannual = balance * rate /2 /100
	set quarterly = balance * rate /4 /100

	output yearly
	output semiannual
	output quarterly

stop

4.

start

	input salary
	
	set incomeTax = salary * .15
	set ssTax = salary * .0765
	set netPay = salary - (incomeTax + ssTax)
	
	output salary
	output incomeTax
	output ssTax
	output netPay

stop
